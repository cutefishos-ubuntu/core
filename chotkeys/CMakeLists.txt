find_package(Qt5 COMPONENTS Core DBus REQUIRED)
find_package(KF5GlobalAccel REQUIRED)

set(PROJECT_SOURCES
    main.cpp
    application.cpp
)

)

    Qt5::Core
    Qt5::DBus
    KF5::GlobalAccel
)
